```ts [useScroll.ts]
import { useEffect, useRef } from 'react'

const useScroll = () => {
  const ref = useRef(null)
  useEffect(() => {
    const handleScroll = () => {
      console.log('scroll')
    }
    ref.current.addEventListener('scroll', handleScroll)
    return () => {
      ref.current.removeEventListener('scroll', handleScroll)
    }
  }, [])
  return ref
}

export default useScroll
```
